TPS65133 개발보드

📢 양음 전원이 필요한 이유는 무엇인가요?

많은 전자 회로는 단일 전원(VCC와 GND)만으로도 충분히 동작합니다. 하지만 OP-AMP나 일부 센서, 액추에이터 같은 아날로그 회로는 양전원(+V)과 음전원(-V)이 모두 필요한 경우가 많습니다. 이런 양음 전원이 필요한 이유는 무엇일까요?


  • ✅ 신호의 양방향 스윙: 0V를 중심으로 양쪽으로 신호가 변화하는 아날로그 신호 처리 가능
  • ✅ 신호 왜곡 감소: DC 바이어스 없이 신호를 처리하여 왜곡 최소화
  • ✅ 다이나믹 레인지 확장: 단일 전원 대비 2배의 신호 범위 확보
  • ✅ OP-AMP 성능 최적화: 대부분의 OP-AMP는 양음 전원에서 최적 성능 발휘
  • ✅ 정밀 센서 구동: 자이로스코프, 로드셀 등 고정밀 센서의 안정적 구동

💡 TPS65133이란 무엇인가요?


TPS65133은 텍사스 인스트루먼트(TI)에서 제조하는 고효율 양음 전원 레귤레이터 IC입니다. 단일 전원에서 양전압과 음전압을 동시에 생성할 수 있어, 이중 전원이 필요한 아날로그 회로 설계에 이상적인 부품입니다.


🌟 TPS65133의 주요 특징
  • 단일 전원에서 양음 전원 생성: 배터리나 USB와 같은 단일 전원으로부터 ±5V 공급 가능
  • 고효율 스위칭 방식: 최대 90% 이상의 전력 변환 효율
  • 낮은 리플: 정밀한 아날로그 회로에 적합한 낮은 출력 리플
  • 소형 패키지: 작은 크기로 공간 제약이 있는 설계에 적합
  • 과전류/과열 보호: 내장된 보호 회로로 안정성 보장

⚡ TPS65133 활용 아두이노 코드 예시

#include <Arduino.h>

// TPS65133 컨트롤 핀 정의
#define TPS65133_EN_PIN 9     // TPS65133 Enable 핀
#define POSITIVE_SENSE_PIN A0  // 양전압 감지 핀
#define NEGATIVE_SENSE_PIN A1  // 음전압 감지 핀
#define LOAD_CONTROL_PIN 10    // 부하 제어 핀

// 전압 모니터링 상수
const float R1 = 10000.0;      // 전압 분배기 상단 저항 (10K)
const float R2 = 10000.0;      // 전압 분배기 하단 저항 (10K)
const float VREF = 3.3;        // ADC 레퍼런스 전압
const float ADC_RESOLUTION = 1023.0; // 10비트 ADC

// 상태 변수
float positiveVoltage = 0.0;
float negativeVoltage = 0.0;
bool powerGood = false;

void setup() {
  Serial.begin(115200);
  
  // 핀 설정
  pinMode(TPS65133_EN_PIN, OUTPUT);
  pinMode(LOAD_CONTROL_PIN, OUTPUT);
  
  // 초기 상태
  digitalWrite(TPS65133_EN_PIN, LOW);  // TPS65133 비활성화
  digitalWrite(LOAD_CONTROL_PIN, LOW); // 부하 비활성화
  
  Serial.println("TPS65133 양음전원 모니터링 시작");
  delay(1000);
  
  // TPS65133 활성화
  digitalWrite(TPS65133_EN_PIN, HIGH);
  Serial.println("TPS65133 활성화 - 안정화 대기 중...");
  delay(100); // 전원 안정화 대기
  
  // 초기 전압 확인
  checkVoltages();
}

void loop() {
  // 전압 측정
  checkVoltages();
  
  // 출력 전압 상태 표시
  Serial.print("양전압: ");
  Serial.print(positiveVoltage, 2);
  Serial.print("V, 음전압: ");
  Serial.print(negativeVoltage, 2);
  Serial.print("V, 상태: ");
  
  if (powerGood) {
    Serial.println("정상");
    digitalWrite(LOAD_CONTROL_PIN, HIGH); // 부하 활성화
  } else {
    Serial.println("비정상 - 전압 확인 필요");
    digitalWrite(LOAD_CONTROL_PIN, LOW);  // 부하 비활성화
  }
  
  // 전원 상태 테스트 (부하 변동 시뮬레이션)
  if (millis() % 10000 < 8000) {  // 8초 켜고 2초 끔
    if (!powerGood) {
      Serial.println("전원 재활성화 시도...");
      digitalWrite(TPS65133_EN_PIN, LOW);
      delay(500);
      digitalWrite(TPS65133_EN_PIN, HIGH);
      delay(100);
    }
  }
  
  delay(1000);  // 1초 간격으로 모니터링
}

void checkVoltages() {
  // ADC 값 읽기
  int positiveRaw = analogRead(POSITIVE_SENSE_PIN);
  int negativeRaw = analogRead(NEGATIVE_SENSE_PIN);
  
  // 전압 계산 (전압 분배기 보정)
  // 양전압은 직접 계산
  positiveVoltage = (positiveRaw / ADC_RESOLUTION) * VREF * ((R1 + R2) / R2);
  
  // 음전압은 반전 계산 필요 (0V 기준이 ADC_RESOLUTION/2)
  float negativeSense = (negativeRaw / ADC_RESOLUTION) * VREF;
  negativeVoltage = -((1.65 - negativeSense) * ((R1 + R2) / R2) * 2);
  
  // 전원 상태 확인 (±5V 기준으로 10% 오차 허용)
  powerGood = (positiveVoltage > 4.5 && positiveVoltage < 5.5 && 
               negativeVoltage < -4.5 && negativeVoltage > -5.5);
}
